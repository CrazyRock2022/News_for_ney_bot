def init_db():
    """–°–æ–∑–¥–∞–µ—Ç —Ç–∞–±–ª–∏—Ü—ã –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π, –∏—Å—Ç–æ—á–Ω–∏–∫–æ–≤ –∏ –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω–Ω—ã—Ö —Å—Å—ã–ª–æ–∫ (–µ—Å–ª–∏ –Ω–µ —Å—É—â–µ—Å—Ç–≤—É—é—Ç)."""
    conn = sqlite3.connect(DB_FILE)
    cur = conn.cursor()
    cur.execute("CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY)")
    cur.execute("CREATE TABLE IF NOT EXISTS sent_links (link TEXT PRIMARY KEY)")
    cur.execute("CREATE TABLE IF NOT EXISTS sources (id INTEGER PRIMARY KEY AUTOINCREMENT, url TEXT NOT NULL UNIQUE)")
    # –ü–µ—Ä–≤–æ–Ω–∞—á–∞–ª—å–Ω–æ–µ –∑–∞–ø–æ–ª–Ω–µ–Ω–∏–µ —Ç–∞–±–ª–∏—Ü—ã sources –∏–∑ —Å–ø–∏—Å–∫–∞ RSS_FEEDS
    cur.execute("SELECT COUNT(*) FROM sources")
    count = cur.fetchone()[0]
    if count == 0:
        for url in RSS_FEEDS:
            cur.execute("INSERT OR IGNORE INTO sources (url) VALUES (?)", (url,))
    conn.commit()
    conn.close()

@dp.message(CommandStart())
async def start_command(message: Message, state: FSMContext):
    """–û–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–æ–º–∞–Ω–¥—ã /start ‚Äî —Ä–µ–≥–∏—Å—Ç—Ä–∏—Ä—É–µ—Ç –Ω–æ–≤–æ–≥–æ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –∏ –æ—Ç–æ–±—Ä–∞–∂–∞–µ—Ç –ø—Ä–∏–≤–µ—Ç—Å—Ç–≤–∏–µ."""
    user_id = message.from_user.id
    # –î–æ–±–∞–≤–ª—è–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –≤ –±–∞–∑—É (–µ—Å–ª–∏ —É–∂–µ –µ—Å—Ç—å, –Ω–∏—á–µ–≥–æ –Ω–µ –ø—Ä–æ–∏–∑–æ–π–¥–µ—Ç)
    conn = sqlite3.connect(DB_FILE)
    cur = conn.cursor()
    cur.execute("INSERT OR IGNORE INTO users (id) VALUES (?)", (user_id,))
    conn.commit()
    conn.close()
    # –ü—Ä–∏–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ —Å –∫–Ω–æ–ø–∫–æ–π "–°–ø—Ä–∞–≤–∫–∞"

        [InlineKeyboardButton(text="–°–ø—Ä–∞–≤–∫–∞", callback_data="help")]

    await message.answer(greeting_text, reply_markup=keyboard)

@dp.message(Command("addsource"))
async def add_source_command(message: Message):
    """–û–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–æ–º–∞–Ω–¥—ã /addsource ‚Äî –¥–æ–±–∞–≤–ª—è–µ—Ç –Ω–æ–≤—ã–π RSS –∏—Å—Ç–æ—á–Ω–∏–∫ –ø–æ URL."""
    # –û–∂–∏–¥–∞–µ—Ç—Å—è, —á—Ç–æ –∫–æ–º–∞–Ω–¥–∞ –≤—ã–∑—ã–≤–∞–µ—Ç—Å—è –≤–º–µ—Å—Ç–µ —Å URL (—á–µ—Ä–µ–∑ –ø—Ä–æ–±–µ–ª)
    args = message.text.split(maxsplit=1)
    if len(args) < 2:
        await message.answer("–ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ: /addsource <URL RSS-–ª–µ–Ω—Ç—ã>")
        return
    url = args[1].strip()
    if not url:
        await message.answer("–ü–æ–∂–∞–ª—É–π—Å—Ç–∞, —É–∫–∞–∂–∏—Ç–µ URL RSS-–ª–µ–Ω—Ç—ã –ø–æ—Å–ª–µ –∫–æ–º–∞–Ω–¥—ã.")
        return
    conn = sqlite3.connect(DB_FILE)
    cur = conn.cursor()
    # –ü—Ä–æ–≤–µ—Ä—è–µ–º, –µ—Å—Ç—å –ª–∏ —É–∂–µ —Ç–∞–∫–æ–π –∏—Å—Ç–æ—á–Ω–∏–∫
    cur.execute("SELECT id FROM sources WHERE url = ?", (url,))
    exists = cur.fetchone()
    if exists:
        conn.close()
        await message.answer("–≠—Ç–æ—Ç –∏—Å—Ç–æ—á–Ω–∏–∫ —É–∂–µ –¥–æ–±–∞–≤–ª–µ–Ω.")
    else:
        try:
            cur.execute("INSERT INTO sources (url) VALUES (?)", (url,))
            conn.commit()
            conn.close()
            await message.answer(f"–ò—Å—Ç–æ—á–Ω–∏–∫ –¥–æ–±–∞–≤–ª–µ–Ω: {html.escape(url)}")
        except Exception as e:
            logging.error(f"Failed to add source {url}: {e}", exc_info=True)
            conn.close()
            await message.answer("–ù–µ —É–¥–∞–ª–æ—Å—å –¥–æ–±–∞–≤–∏—Ç—å –∏—Å—Ç–æ—á–Ω–∏–∫. –£–±–µ–¥–∏—Ç–µ—Å—å, —á—Ç–æ URL —É–∫–∞–∑–∞–Ω –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ.")

@dp.message(Command("removesource"))
async def remove_source_command(message: Message):
    """–û–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–æ–º–∞–Ω–¥—ã /removesource ‚Äî —É–¥–∞–ª—è–µ—Ç —É–∫–∞–∑–∞–Ω–Ω—ã–π RSS –∏—Å—Ç–æ—á–Ω–∏–∫ (–ø–æ URL –∏–ª–∏ ID)."""
    args = message.text.split(maxsplit=1)
    if len(args) < 2:
        await message.answer("–ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ: /removesource <URL –∏–ª–∏ ID –∏—Å—Ç–æ—á–Ω–∏–∫–∞>")
        return
    target = args[1].strip()
    conn = sqlite3.connect(DB_FILE)
    cur = conn.cursor()
    if target.isdigit():
        source_id = int(target)
        # –£–¥–∞–ª—è–µ–º –ø–æ ID
        cur.execute("SELECT url FROM sources WHERE id = ?", (source_id,))
        row = cur.fetchone()
        if not row:
            conn.close()
            await message.answer("–ò—Å—Ç–æ—á–Ω–∏–∫ —Å —É–∫–∞–∑–∞–Ω–Ω—ã–º ID –Ω–µ –Ω–∞–π–¥–µ–Ω.")
            return
        url_to_remove = row[0]
        cur.execute("DELETE FROM sources WHERE id = ?", (source_id,))
        conn.commit()
        conn.close()
        await message.answer(f"–ò—Å—Ç–æ—á–Ω–∏–∫ —É–¥–∞–ª—ë–Ω: {html.escape(url_to_remove)}")
    else:
        url = target
        cur.execute("DELETE FROM sources WHERE url = ?", (url,))
        changes = cur.rowcount
        conn.commit()
        conn.close()
        if changes:
            await message.answer(f"–ò—Å—Ç–æ—á–Ω–∏–∫ —É–¥–∞–ª—ë–Ω: {html.escape(url)}")
        else:
            await message.answer("–£–∫–∞–∑–∞–Ω–Ω—ã–π –∏—Å—Ç–æ—á–Ω–∏–∫ –Ω–µ –Ω–∞–π–¥–µ–Ω –≤ –±–∞–∑–µ.")

@dp.message(Command("listsources"))
async def list_sources_command(message: Message):
    """–û–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–æ–º–∞–Ω–¥—ã /listsources ‚Äî –≤—ã–≤–æ–¥–∏—Ç —Å–ø–∏—Å–æ–∫ –≤—Å–µ—Ö —Ç–µ–∫—É—â–∏—Ö RSS-–∏—Å—Ç–æ—á–Ω–∏–∫–æ–≤."""
    conn = sqlite3.connect(DB_FILE)
    cur = conn.cursor()
    cur.execute("SELECT id, url FROM sources")
    sources_list = cur.fetchall()
    conn.close()
    if not sources_list:
        await message.answer("–°–ø–∏—Å–æ–∫ –∏—Å—Ç–æ—á–Ω–∏–∫–æ–≤ –ø—É—Å—Ç. –î–æ–±–∞–≤—å—Ç–µ –Ω–æ–≤—ã–π –∏—Å—Ç–æ—á–Ω–∏–∫ –∫–æ–º–∞–Ω–¥–æ–π /addsource.")
    else:
        response_lines = [hbold("–¢–µ–∫—É—â–∏–µ –∏—Å—Ç–æ—á–Ω–∏–∫–∏:")]
        for source_id, url in sources_list:
            response_lines.append(f"{source_id}. {html.escape(url)}")
        response_text = "\n".join(response_lines)
        await message.answer(response_text)

# ------- GPT –ê–ù–ê–õ–ò–ó -------
async def gpt_check(prompt: str) -> str:
    """–û—Ç–ø—Ä–∞–≤–ª—è–µ—Ç –∑–∞–ø—Ä–æ—Å –∫ OpenAI GPT-4 –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –æ—Ç–≤–µ—Ç –º–æ–¥–µ–ª–∏ ('–¥–∞' –∏–ª–∏ '–Ω–µ—Ç')."""
    try:
        response = await openai.ChatCompletion.acreate(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            max_tokens=5,
            temperature=0
        )
        answer = response.choices[0].message.content.strip().lower()
        return answer
    except Exception as e:
        logging.error(f"OpenAI API error: {e}", exc_info=True)
        # –í —Å–ª—É—á–∞–µ –æ—à–∏–±–∫–∏ API —Å—á–∏—Ç–∞–µ–º –Ω–æ–≤–æ—Å—Ç—å –Ω–µ—Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω–æ–π
        return "–Ω–µ—Ç"

# ------- –û–ë–†–ê–ë–û–¢–ö–ê –ù–û–í–û–°–¢–ï–ô -------
async def fetch_news(prompt: str):
    """–ü–∞—Ä—Å–∏—Ç –≤—Å–µ RSS-–∏—Å—Ç–æ—á–Ω–∏–∫–∏ –∏ –∞–Ω–∞–ª–∏–∑–∏—Ä—É–µ—Ç –Ω–æ–≤–æ—Å—Ç–∏ —Å –ø–æ–º–æ—â—å—é GPT.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∫–æ—Ä—Ç–µ–∂: (—Å–ø–∏—Å–æ–∫ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö –Ω–æ–≤–æ—Å—Ç–µ–π, —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –ø–æ –∏—Å—Ç–æ—á–Ω–∏–∫–∞–º)."""
    conn = sqlite3.connect(DB_FILE)
    cur = conn.cursor()
    # –ü–æ–ª—É—á–∞–µ–º —É–∂–µ –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω–Ω—ã–µ —Å—Å—ã–ª–∫–∏, —á—Ç–æ–±—ã –Ω–µ –¥—É–±–ª–∏—Ä–æ–≤–∞—Ç—å
    cur.execute("SELECT link FROM sent_links")
    sent_links_set = {row[0] for row in cur.fetchall()}
    relevant_articles = []   # —Å–ø–∏—Å–æ–∫ —Å–æ–æ–±—â–µ–Ω–∏–π —Å —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–º–∏ –Ω–æ–≤–æ—Å—Ç—è–º–∏ (–¥–ª—è –æ—Ç–ø—Ä–∞–≤–∫–∏)
    stats_by_source = {}     # —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –ø–æ –∫–∞–∂–¥–æ–º—É –∏—Å—Ç–æ—á–Ω–∏–∫—É

    now = datetime.now()
    one_week_ago = now - timedelta(days=7)

    # –ü–æ–ª—É—á–∞–µ–º –≤—Å–µ –∏—Å—Ç–æ—á–Ω–∏–∫–∏ –∏–∑ –±–∞–∑—ã –¥–∞–Ω–Ω—ã—Ö –∏ –ø–∞—Ä—Å–∏–º –∏—Ö RSS
    cur.execute("SELECT url FROM sources")
    sources = [row[0] for row in cur.fetchall()]
    for url in sources:
        feed = feedparser.parse(url)
        # –ù–∞–∑–≤–∞–Ω–∏–µ –∏—Å—Ç–æ—á–Ω–∏–∫–∞ –¥–ª—è –æ—Ç—á—ë—Ç–∞ (–µ—Å–ª–∏ –µ—Å—Ç—å, –∏–Ω–∞—á–µ –∏—Å–ø–æ–ª—å–∑—É–µ–º URL)
        source_title = feed.feed.get("title", url) if hasattr(feed, 'feed') else url
        total_count = 0
        relevant_count = 0
        possible_count = 0
        not_relevant_count = 0

        for entry in feed.entries:
            link = entry.get("link", "")
            # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º, –µ—Å–ª–∏ –Ω–µ—Ç —Å—Å—ã–ª–∫–∏ –∏–ª–∏ —É–∂–µ –æ—Ç–ø—Ä–∞–≤–ª—è–ª–∏ —ç—Ç—É –Ω–æ–≤–æ—Å—Ç—å —Ä–∞–Ω–µ–µ
            if not link or link in sent_links_set:
                continue
            # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º –Ω–æ–≤–æ—Å—Ç–∏ —Å—Ç–∞—Ä—à–µ 7 –¥–Ω–µ–π (–µ—Å–ª–∏ –¥–∞—Ç–∞ —É–∫–∞–∑–∞–Ω–∞)
            entry_date = None
            if 'published_parsed' in entry and entry.published_parsed:
                try:
                    entry_date = datetime.fromtimestamp(time.mktime(entry.published_parsed))
                except Exception:
                    entry_date = None
            if entry_date and entry_date < one_week_ago:
                continue

            total_count += 1
            title = entry.get("title", "(–±–µ–∑ –∑–∞–≥–æ–ª–æ–≤–∫–∞)")
            summary = entry.get("summary", "")
            category = entry.get("category", "")
            tags = []
            if "tags" in entry:
                tags = [t.get("term", "") for t in entry.tags]
            content = ""
            if "content" in entry and entry.content:
                content = entry.content[0].get("value", "")

            # –§–æ—Ä–º–∏—Ä—É–µ–º –ø–æ–ª–Ω—ã–π —Ç–µ–∫—Å—Ç –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞ (–∑–∞–≥–æ–ª–æ–≤–æ–∫, –æ–ø–∏—Å–∞–Ω–∏–µ, –∫–∞—Ç–µ–≥–æ—Ä–∏—è, —Ç–µ–≥–∏, –∫–æ–Ω—Ç–µ–Ω—Ç)
            full_text = (
                f"–ó–∞–≥–æ–ª–æ–≤–æ–∫: {title}\n"
                f"–û–ø–∏—Å–∞–Ω–∏–µ: {summary}\n"
                f"–ö–∞—Ç–µ–≥–æ—Ä–∏—è: {category}\n"
                f"–¢–µ–≥–∏: {', '.join(tags)}\n"
                f"–ö–æ–Ω—Ç–µ–Ω—Ç: {content}"
            )
            # –ü—Ä–æ–º—Ç –¥–ª—è GPT (–æ–±—ä–µ–¥–∏–Ω—è–µ–º –ø—Ä–æ–º—Ç –∏ —Ç–µ–∫—Å—Ç –Ω–æ–≤–æ—Å—Ç–∏, —Ç—Ä–µ–±—É–µ–º –æ—Ç–≤–µ—Ç –î–∞/–ù–µ—Ç)
            gpt_prompt = f"{prompt}\n\n{full_text}\n\n–î–∞ –∏–ª–∏ –ù–µ—Ç?"
            answer = await gpt_check(gpt_prompt)
            # –ö–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º –æ—Ç–≤–µ—Ç: "–¥–∞" (—Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω–æ), "–Ω–µ—Ç" (–Ω–µ—Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω–æ), –∏–Ω–∞—á–µ ‚Äî –≤–æ–∑–º–æ–∂–Ω–æ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω–æ
            if "–¥–∞" in answer or "yes" in answer:
                relevant_count += 1
                # –§–æ—Ä–º–∏—Ä—É–µ–º —Å–æ–æ–±—â–µ–Ω–∏–µ –¥–ª—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è: –∑–∞–≥–æ–ª–æ–≤–æ–∫ (bold) –∏ —Å—Å—ã–ª–∫–∞
                relevant_articles.append(f"<b>{title}</b>\n{link}")
                # –ü–æ–º–µ—á–∞–µ–º —Å—Å—ã–ª–∫—É –∫–∞–∫ –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω–Ω—É—é
                sent_links_set.add(link)
            elif "–Ω–µ—Ç" in answer or "no" in answer:
                not_relevant_count += 1
            else:
                possible_count += 1

        # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É –ø–æ —Ç–µ–∫—É—â–µ–º—É –∏—Å—Ç–æ—á–Ω–∏–∫—É
        stats_by_source[source_title] = {
            "total": total_count,
            "relevant": relevant_count,
            "possible": possible_count,
            "not_relevant": not_relevant_count
        }

    # –î–æ–±–∞–≤–ª—è–µ–º –Ω–æ–≤—ã–µ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–µ —Å—Å—ã–ª–∫–∏ –≤ –±–∞–∑—É (—Ç–∞–±–ª–∏—Ü–∞ sent_links)
    for article in relevant_articles:
        link = article.split("\n")[-1]
        cur.execute("INSERT OR IGNORE INTO sent_links (link) VALUES (?)", (link,))
    conn.commit()
    conn.close()
    return relevant_articles, stats_by_source

# ------- –ö–û–ú–ê–ù–î–ê /digest -------
@dp.message(Command("digest"))
async def digest_command(message: Message, state: FSMContext):
    """–û–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–æ–º–∞–Ω–¥—ã /digest ‚Äî —Å–æ–±–∏—Ä–∞–µ—Ç –Ω–æ–≤–æ—Å—Ç–∏ –∏ –≤—ã–≤–æ–¥–∏—Ç –æ—Ç—á—ë—Ç –ø–æ –∏—Å—Ç–æ—á–Ω–∏–∫–∞–º."""
    await message.answer("‚ö°Ô∏è –°–æ–±–∏—Ä–∞—é —Å–≤–µ–∂–∏–µ –Ω–æ–≤–æ—Å—Ç–∏, –ø–æ–∂–∞–ª—É–π—Å—Ç–∞ –ø–æ–¥–æ–∂–¥–∏—Ç–µ...")
    # –ê–Ω–∞–ª–∏–∑–∏—Ä—É–µ–º –Ω–æ–≤–æ—Å—Ç–∏ —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º –¥–µ—Ñ–æ–ª—Ç–Ω–æ–≥–æ –ø—Ä–æ–º—Ç–∞
    relevant_articles, stats = await fetch_news(DEFAULT_PROMPT)
    # –°—É–º–º–∏—Ä—É–µ–º –æ–±—â–∏–µ –ø–æ–∫–∞–∑–∞—Ç–µ–ª–∏ –ø–æ –≤—Å–µ–º –∏—Å—Ç–æ—á–Ω–∏–∫–∞–º
    total_viewed = sum(data["total"] for data in stats.values())
    total_relevant = sum(data["relevant"] for data in stats.values())
    total_possible = sum(data["possible"] for data in stats.values())
    total_not_rel = sum(data["not_relevant"] for data in stats.values())
    # –ï—Å–ª–∏ –Ω–∞–π–¥–µ–Ω—ã —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–µ –Ω–æ–≤–æ—Å—Ç–∏
    if total_relevant > 0:
        # –§–æ—Ä–º–∏—Ä—É–µ–º —Ç–µ–∫—Å—Ç –æ—Ç—á—ë—Ç–∞ –ø–æ –∫–∞–∂–¥–æ–º—É –∏—Å—Ç–æ—á–Ω–∏–∫—É
        report_lines = ["<b>–û—Ç—á—ë—Ç –ø–æ –∏—Å—Ç–æ—á–Ω–∏–∫–∞–º:</b>"]
        for source_title, data in stats.items():
            report_lines.append(
                f"{source_title}: –ø—Ä–æ—Å–º–æ—Ç—Ä–µ–Ω–æ {data['total']}, "
                f"—Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö {data['relevant']}, "
                f"–≤–æ–∑–º–æ–∂–Ω–æ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö {data['possible']}, "
                f"–Ω–µ—Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö {data['not_relevant']}"
            )
        report_lines.append(
            f"\n<b>–û–±—â–∏–π –∏—Ç–æ–≥:</b> –ø—Ä–æ—Å–º–æ—Ç—Ä–µ–Ω–æ {total_viewed}, "
            f"—Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö {total_relevant}, "
            f"–≤–æ–∑–º–æ–∂–Ω–æ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö {total_possible}, "
            f"–Ω–µ—Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö {total_not_rel}."
        )
        digest_text = "\n".join(report_lines)
        # –ö–Ω–æ–ø–∫–∞ –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è —Å–ø–∏—Å–∫–∞ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö –Ω–æ–≤–æ—Å—Ç–µ–π
        keyboard = InlineKeyboardMarkup(inline_keyboard=[
            [InlineKeyboardButton(text="üìÑ –ü–æ–ª—É—á–∏—Ç—å –Ω–æ–≤–æ—Å—Ç–∏", callback_data="get_news")]
        ])
        # –û—Ç–ø—Ä–∞–≤–ª—è–µ–º –æ—Ç—á—ë—Ç —Å –∫–Ω–æ–ø–∫–æ–π
        await message.answer(digest_text, reply_markup=keyboard)
        # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Å–ø–∏—Å–æ–∫ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö –Ω–æ–≤–æ—Å—Ç–µ–π –≤–æ –≤—Ä–µ–º–µ–Ω–Ω–æ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
        await state.update_data(relevant_news_list=relevant_articles)
    else:
        # –ï—Å–ª–∏ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö –Ω–æ–≤–æ—Å—Ç–µ–π –Ω–µ –Ω–∞–π–¥–µ–Ω–æ ‚Äî –ø—Ä–µ–¥–ª–∞–≥–∞–µ–º –≤–≤–µ—Å—Ç–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å—Å–∫–∏–π –∑–∞–ø—Ä–æ—Å
        await message.answer("–†–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö –Ω–æ–≤–æ—Å—Ç–µ–π –Ω–µ –Ω–∞–π–¥–µ–Ω–æ.\n–í–≤–µ–¥–∏—Ç–µ —Å–≤–æ–π –∑–∞–ø—Ä–æ—Å –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞ –Ω–æ–≤–æ—Å—Ç–µ–π:")
        await state.set_state(UserState.awaiting_prompt)

@dp.message(UserState.awaiting_prompt)
async def handle_user_prompt(message: Message, state: FSMContext):
    """–û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å—Å–∫–∏–π –ø—Ä–æ–º—Ç –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞ –Ω–æ–≤–æ—Å—Ç–µ–π (–µ—Å–ª–∏ –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é –Ω–µ –Ω–∞–π–¥–µ–Ω–æ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö)."""
    prompt = message.text.strip()
    await message.answer("üîé –í—ã–ø–æ–ª–Ω—è—é –∞–Ω–∞–ª–∏–∑ –Ω–æ–≤–æ—Å—Ç–µ–π –ø–æ –≤–∞—à–µ–º—É –∑–∞–ø—Ä–æ—Å—É, –ø–æ–∂–∞–ª—É–π—Å—Ç–∞ –ø–æ–¥–æ–∂–¥–∏—Ç–µ...")
    # –ü–æ–≤—Ç–æ—Ä–Ω–æ –∞–Ω–∞–ª–∏–∑–∏—Ä—É–µ–º –≤—Å–µ –Ω–æ–≤–æ—Å—Ç–∏ —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å—Å–∫–æ–≥–æ –ø—Ä–æ–º—Ç–∞
    relevant_articles, stats = await fetch_news(prompt)
    if relevant_articles:
        # –û—Ç–ø—Ä–∞–≤–ª—è–µ–º –∫–∞–∂–¥—É—é –Ω–∞–π–¥–µ–Ω–Ω—É—é —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—É—é –Ω–æ–≤–æ—Å—Ç—å –æ—Ç–¥–µ–ª—å–Ω—ã–º —Å–æ–æ–±—â–µ–Ω–∏–µ–º
        for article in relevant_articles:
            await message.answer(article)
    else:
        await message.answer("–ü–æ –≤–∞—à–µ–º—É –∑–∞–ø—Ä–æ—Å—É —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã—Ö –Ω–æ–≤–æ—Å—Ç–µ–π –Ω–µ –Ω–∞–π–¥–µ–Ω–æ.")
    # –°–±—Ä–∞—Å—ã–≤–∞–µ–º —Å–æ—Å—Ç–æ—è–Ω–∏–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
    await state.clear()

# ------- CALLBACK: –ü–û–õ–£–ß–ï–ù–ò–ï –ù–û–í–û–°–¢–ï–ô -------
@dp.callback_query(F.data == "get_news")
async def get_news_callback(callback: CallbackQuery, state: FSMContext):
    """–û–±—Ä–∞–±–æ—Ç—á–∏–∫ –Ω–∞–∂–∞—Ç–∏—è –∫–Ω–æ–ø–∫–∏ '–ü–æ–ª—É—á–∏—Ç—å –Ω–æ–≤–æ—Å—Ç–∏' ‚Äî –≤—ã—Å—ã–ª–∞–µ—Ç —Å–æ—Ö—Ä–∞–Ω—ë–Ω–Ω—ã–µ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–µ –Ω–æ–≤–æ—Å—Ç–∏."""
    user_data = await state.get_data()
    news_list = user_data.get("relevant_news_list", [])
    if not news_list:
        await callback.answer("–ù–µ—Ç –Ω–æ–≤–æ—Å—Ç–µ–π –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è.", show_alert=True)
        return
    # –û—Ç–ø—Ä–∞–≤–ª—è–µ–º –∫–∞–∂–¥—É—é —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—É—é –Ω–æ–≤–æ—Å—Ç—å –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—é
    for article in news_list:
        await callback.message.answer(article)
    # –£–±–∏—Ä–∞–µ–º —É –∏—Å—Ö–æ–¥–Ω–æ–≥–æ —Å–æ–æ–±—â–µ–Ω–∏—è –∫–Ω–æ–ø–∫—É, —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å –ø–æ–≤—Ç–æ—Ä–Ω–æ–≥–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
    await callback.message.edit_reply_markup(reply_markup=None)
    await callback.answer()  # –ó–∞–∫—Ä—ã–≤–∞–µ–º –≤—Å–ø–ª—ã–≤–∞—é—â–µ–µ —É–≤–µ–¥–æ–º–ª–µ–Ω–∏–µ
    # –û—á–∏—â–∞–µ–º —Å–ø–∏—Å–æ–∫ —Å–æ—Ö—Ä–∞–Ω—ë–Ω–Ω—ã—Ö –Ω–æ–≤–æ—Å—Ç–µ–π –∏–∑ —Å–æ—Å—Ç–æ—è–Ω–∏—è
    await state.update_data(relevant_news_list=[])

# ------- –ï–ñ–ï–î–ù–ï–í–ù–ê–Ø –†–ê–°–°–´–õ–ö–ê -------
async def scheduled_job():
    """–ï–∂–µ–¥–Ω–µ–≤–Ω–∞—è –∑–∞–¥–∞—á–∞ (–≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –≤ 11:00) ‚Äî —Å–æ–±–∏—Ä–∞–µ—Ç –Ω–æ–≤–æ—Å—Ç–∏ –∏ —Ä–∞—Å—Å—ã–ª–∞–µ—Ç —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–µ –≤—Å–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è–º."""
    logging.info("Scheduled job started: fetching and sending news...")
    try:
        # –ê–Ω–∞–ª–∏–∑–∏—Ä—É–µ–º –Ω–æ–≤–æ—Å—Ç–∏ —Å –¥–µ—Ñ–æ–ª—Ç–Ω—ã–º –ø—Ä–æ–º—Ç–æ–º
        relevant_articles, stats = await fetch_news(DEFAULT_PROMPT)
        # –ï—Å–ª–∏ –ø–æ—è–≤–∏–ª–∏—Å—å –Ω–æ–≤—ã–µ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–µ —Å—Ç–∞—Ç—å–∏, —Ä–∞—Å—Å—ã–ª–∞–µ–º –∏—Ö –≤—Å–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è–º
        if relevant_articles:
            conn = sqlite3.connect(DB_FILE)
            cur = conn.cursor()
            cur.execute("SELECT id FROM users")
            user_ids = [row[0] for row in cur.fetchall()]
            conn.close()
            # –†–∞—Å—Å—ã–ª–∞–µ–º –∫–∞–∂–¥–æ–º—É –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—é –≤—Å–µ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω—ã–µ –Ω–æ–≤–æ—Å—Ç–∏
            for user_id in user_ids:
                for article in relevant_articles:
                    try:
                        await bot.send_message(user_id, article)
                    except Exception as e:
                        # –õ–æ–≥–∏—Ä—É–µ–º –æ—à–∏–±–∫—É –æ—Ç–ø—Ä–∞–≤–∫–∏ (–Ω–∞–ø—Ä–∏–º–µ—Ä, –µ—Å–ª–∏ –±–æ—Ç –∑–∞–±–ª–æ–∫–∏—Ä–æ–≤–∞–Ω –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–º)
                        logging.error(f"Failed to send news to user {user_id}: {e}", exc_info=True)
                        # –ï—Å–ª–∏ –±–æ—Ç –∑–∞–±–ª–æ–∫–∏—Ä–æ–≤–∞–Ω –∏–ª–∏ —á–∞—Ç –Ω–µ–¥–æ—Å—Ç—É–ø–µ–Ω ‚Äî —É–¥–∞–ª—è–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –∏–∑ –±–∞–∑—ã
                        if "Forbidden" in str(e) or "blocked" in str(e):
                            conn2 = sqlite3.connect(DB_FILE)
                            cur2 = conn2.cursor()
                            cur2.execute("DELETE FROM users WHERE id = ?", (user_id,))
                            conn2.commit()
                            conn2.close()
            logging.info(f"Scheduled job: sent {len(relevant_articles)} news to {len(user_ids)} users.")
        else:
            logging.info("Scheduled job: no relevant news found to send.")
    except Exception as e:
        logging.error(f"Error in scheduled job: {e}", exc_info=True)

# ------- –ó–ê–ü–£–°–ö –ü–û–õ–õ–ò–ù–ì–ê -------
async def main():
    # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –±–∞–∑—ã –¥–∞–Ω–Ω—ã—Ö
    init_db()

    scheduler.add_job(scheduled_job, "cron", hour=11, minute=0)
    scheduler.start()

    await bot.delete_webhook(drop_pending_updates=True)

    await dp.start_polling(bot)
